-- 1. Add tracking for the "Remind Admin" rate limit
ALTER TABLE users ADD COLUMN IF NOT EXISTS last_reminded_at TIMESTAMP WITH TIME ZONE;

-- 2. Delete old function version to allow return type change
DROP FUNCTION IF EXISTS verify_user_and_reward(bigint, bigint, integer);

-- 3. Create the new idempotent version (prevents double payouts)
CREATE OR REPLACE FUNCTION verify_user_and_reward(u_id BIGINT, r_id BIGINT, amt INTEGER)
RETURNS BOOLEAN AS $$
DECLARE
    rows_affected INTEGER;
BEGIN
    -- Try to activate ONLY if currently unverified
    UPDATE users SET 
        is_verified = TRUE, 
        state = 'idle' 
    WHERE telegram_id = u_id AND is_verified = FALSE;
    
    GET DIAGNOSTICS rows_affected = ROW_COUNT;

    -- Only reward if the update actually happened
    IF rows_affected > 0 THEN
        IF r_id IS NOT NULL THEN
            UPDATE users SET 
                balance = balance + amt, 
                total_referrals = total_referrals + 1 
            WHERE telegram_id = r_id;
        END IF;
        RETURN TRUE;
    END IF;

    RETURN FALSE;
END;
$$ LANGUAGE plpgsql;
